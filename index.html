ÍNDICES para performance -- ============================================================ CREATE INDEX IF NOT EXISTS idx_progresso_cliente_id ON checklist_progresso (cliente_id); CREATE INDEX IF NOT EXISTS idx_progresso_marcado_por ON checklist_progresso (marcado_por); CREATE INDEX IF NOT EXISTS idx_progresso_concluido ON checklist_progresso (concluido); CREATE INDEX IF NOT EXISTS idx_progresso_grupo ON checklist_progresso (grupo); CREATE INDEX IF NOT EXISTS idx_clientes_funcionario ON clientes (funcionario); CREATE INDEX IF NOT EXISTS idx_clientes_tipo ON clientes (tipo); -- ============================================================ -- 6. VIEW: vw_progresso_clientes -- ============================================================ CREATE OR REPLACE VIEW vw_progresso_clientes AS SELECT c.id, c.nome, c.tipo, c.funcionario, c.data, c.criado_em, CASE WHEN LOWER(c.tipo) LIKE '%retaguarda%' THEN 18 WHEN LOWER(c.tipo) LIKE '%nfce%' THEN 16 WHEN LOWER(c.tipo) LIKE '%nf-ce%' THEN 16 ELSE 22 END AS total_steps, COUNT(p.id) FILTER (WHERE p.concluido = TRUE) AS steps_concluidos, ROUND( COUNT(p.id) FILTER (WHERE p.concluido = TRUE)::NUMERIC / NULLIF(CASE WHEN LOWER(c.tipo) LIKE '%retaguarda%' THEN 18 WHEN LOWER(c.tipo) LIKE '%nfce%' THEN 16 WHEN LOWER(c.tipo) LIKE '%nf-ce%' THEN 16 ELSE 22 END, 0) * 100 , 1) AS percentual FROM clientes c LEFT JOIN checklist_progresso p ON p.cliente_id = c.id GROUP BY c.id, c.nome, c.tipo, c.funcionario, c.data, c.criado_em ORDER BY c.criado_em DESC; -- ============================================================ -- 7. VIEW: vw_ranking_colaboradores -- ============================================================ CREATE OR REPLACE VIEW vw_ranking_colaboradores AS SELECT p.marcado_por AS colaborador, COUNT(*) FILTER (WHERE p.concluido = TRUE) AS total_concluidos, COUNT(DISTINCT p.cliente_id) AS total_clientes, COUNT(*) FILTER (WHERE p.concluido = TRUE AND p.grupo = 'servidor') AS servidor, COUNT(*) FILTER (WHERE p.concluido = TRUE AND p.grupo = 'retaguarda') AS retaguarda, COUNT(*) FILTER (WHERE p.concluido = TRUE AND p.grupo = 'nfce') AS nfce, ROUND( COUNT(*) FILTER (WHERE p.concluido = TRUE)::NUMERIC / NULLIF(COUNT(DISTINCT p.cliente_id), 0) , 1) AS media_por_cliente, MAX(p.marcado_em) AS ultima_atividade FROM checklist_progresso p WHERE p.marcado_por IS NOT NULL AND p.marcado_por <> '' AND p.marcado_por <> '—' GROUP BY p.marcado_por ORDER BY total_concluidos DESC; -- ============================================================ -- 8. VIEW: vw_resumo_geral -- ============================================================ CREATE OR REPLACE VIEW vw_resumo_geral AS SELECT COUNT(DISTINCT c.id) AS total_clientes, COUNT(p.id) FILTER (WHERE p.concluido = TRUE) AS total_concluidos, COUNT(p.id) FILTER (WHERE p.concluido = FALSE OR p.id IS NULL) AS total_pendentes, ROUND( COUNT(p.id) FILTER (WHERE p.concluido = TRUE)::NUMERIC / NULLIF(COUNT(p.id), 0) * 100 , 1) AS percentual_geral FROM clientes c LEFT JOIN checklist_progresso p ON p.cliente_id = c.id; -- ============================================================ -- VERIFICAÇÃO FINAL — rode após o script para confirmar -- ============================================================ -- SELECT column_name, data_type -- FROM information_schema.columns -- WHERE table_name = 'checklist_progresso' -- ORDER BY ordinal_position; -- SELECT * FROM vw_resumo_geral; -- SELECT * FROM vw_ranking_colaboradores; -- SELECT * FROM vw_progresso_clientes LIMIT 10;
